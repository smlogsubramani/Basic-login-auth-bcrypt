app.post('/user', async (req,res)=>{
  try{
    const salt = await bcrypt.genSalt()
    const hashedpass = await bcrypt.hash(req.body.password , salt)
    console.log(salt)
    console.log(hashedpass)

    const newUser = new User({
      name: req.body.name,
      password: hashedpass
    });

    await newUser.save();
    res.status(201).send();
  }catch{
    res.status(500).send();
  }
});
